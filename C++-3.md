







静态存储区：



```c++
#include<iostream>
using namespace std;


int main(){
    
    
    static int a;//全局静态变量
    
    
    void print(int& a){
        static int b;//局部静态变量
        cout<<b<<endl;
    }
    
    static int sum(int& a, in& b){//静态函数
        return a+b;
    }
    
    animal::eat();
    

    }
    
    
    return 0;
}

class animal{
public:
    animal(){

    };

    static weight;//类的静态成员，不再属于对象

    static void eat(){//类的静态函数，不再属于对象
        //不可以引用类中的非静态成员
        cout<<"eat"<<endl;
    }
```
